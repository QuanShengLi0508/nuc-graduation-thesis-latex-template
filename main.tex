\newcommand{\SetupSongtiFont}{%
  \IfFileExists{fonts/local/simsun.ttc}{%
    \setCJKmainfont[Path=fonts/local/]{simsun.ttc}%
    \setCJKfamilyfont{zhsong}[Path=fonts/local/]{simsun.ttc}%
  }{%
    \IfFileExists{fonts/local/songti.ttc}{%
      \setCJKmainfont[Path=fonts/local/,FontIndex=4]{songti.ttc}%
      \setCJKfamilyfont{zhsong}[Path=fonts/local/,FontIndex=4]{songti.ttc}%
    }{%
      \IfFontExistsTF{SimSun}{%
        \setCJKmainfont{SimSun}%
        \setCJKfamilyfont{zhsong}{SimSun}%
      }{%
        \IfFontExistsTF{宋体}{%
          \setCJKmainfont{宋体}%
          \setCJKfamilyfont{zhsong}{宋体}%
        }{%
          \IfFontExistsTF{STSong}{%
            \setCJKmainfont{STSong}%
            \setCJKfamilyfont{zhsong}{STSong}%
          }{%
            \IfFontExistsTF{Songti SC}{%
              \setCJKmainfont{Songti SC}%
              \setCJKfamilyfont{zhsong}{Songti SC}%
            }{%
              \IfFontExistsTF{Source Han Serif SC}{%
                \setCJKmainfont{Source Han Serif SC}%
                \setCJKfamilyfont{zhsong}{Source Han Serif SC}%
                \TemplateFontWarning{Using Source Han Serif SC as a preview fallback. Install SimSun/STSong for strict Songti output.}%
              }{%
                \PackageError{thesis-template}{No Songti-like font found}{Install SimSun/STSong/Songti SC, or put simsun.ttc or songti.ttc into fonts/local/.}%
              }%
            }%
          }%
        }%
      }%
    }%
  }%
}
